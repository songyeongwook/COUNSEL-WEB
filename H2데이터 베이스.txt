상담사
CREATE TABLE COUNSEL (
    COUNSEL_ID VARCHAR(50) PRIMARY KEY,
    PASSWORD VARCHAR(100) NOT NULL,
    NAME VARCHAR(100) NOT NULL
);

고객
CREATE TABLE USERS (
    USER_ID VARCHAR(100) PRIMARY KEY,
    PASSWORD VARCHAR(100) NOT NULL,
    NAME VARCHAR(100) NOT NULL
);

-피드백 파일
CREATE TABLE COUNSEL_FILES (
    FILE_ID INT AUTO_INCREMENT PRIMARY KEY,
    COUNSEL_NUM INT,
    COUNSEL_ID VARCHAR(100),
    USER_ID VARCHAR(100),
    FILE_NAME VARCHAR(255),
    FILE_PATH VARCHAR(500),
    UPLOAD_DATE TIMESTAMP,

    FOREIGN KEY (COUNSEL_ID) REFERENCES COUNSEL(COUNSEL_ID)
    FOREIGN KEY (USER_ID) REFERENCES USERS(USER_ID)
);

-상담내역
CREATE TABLE COUNSEL_HISTORY (
    COUNSEL_NUM INT AUTO_INCREMENT PRIMARY KEY,
    USER_ID VARCHAR(50),
    COUNSEL_ID VARCHAR(50),
    TITLE VARCHAR(255),
    CONTENT VARCHAR,
    COUNSEL_DATE TIMESTAMP,
    CONFIRMED CHAR(1) DEFAULT 'N',

    FOREIGN KEY (USER_ID) REFERENCES USERS(USER_ID)        
    FOREIGN KEY (COUNSEL_ID) REFERENCES COUNSEL(COUNSEL_ID)
);

-공지사항
CREATE TABLE COUNSEL_NOTI (
    NOTI_NUM INT AUTO_INCREMENT PRIMARY KEY,
    COUNSEL_ID VARCHAR(50),
    TITLE VARCHAR(200),
    CONTEXT VARCHAR,

    FOREIGN KEY (COUNSEL_ID) REFERENCES COUNSEL(COUNSEL_ID)
);
